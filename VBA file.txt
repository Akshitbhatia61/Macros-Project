' Removespaceandsort Macro
' To remove the spaces in the data and sort them
'
' Keyboard Shortcut: Ctrl+Shift+L

    Columns("D:I").Select
    ActiveWorkbook.Worksheets("Colorado List").Sort.SortFields.Clear
    ActiveWorkbook.Worksheets("Colorado List").Sort.SortFields.Add2 Key:=Range( _
        "D1:D2407"), SortOn:=xlSortOnValues, Order:=xlAscending, DataOption:= _
        xlSortNormal
    With ActiveWorkbook.Worksheets("Colorado List").Sort
        .SetRange Range("D1:I2407")
        .Header = xlGuess
        .MatchCase = False
        .Orientation = xlTopToBottom
        .SortMethod = xlPinYin
        .Apply
    End With
    Range("D1:I1").Select
    Selection.Insert Shift:=xlDown, CopyOrigin:=xlFormatFromLeftOrAbove
    Range("A1:A6").Select
    Selection.Copy
    Range("D1").Select
    Selection.PasteSpecial Paste:=xlPasteAll, Operation:=xlNone, SkipBlanks:= _
        False, Transpose:=True
    Columns("D:I").Select
    Columns("D:I").EntireColumn.AutoFit
    Range("D1:I1").Select
    Application.CutCopyMode = False
    With Selection.Interior
        .Pattern = xlSolid
        .PatternColorIndex = xlAutomatic
        .ThemeColor = xlThemeColorAccent1
        .TintAndShade = 0.399975585192419
        .PatternTintAndShade = 0
    End With
End Sub
